# ===========================================
# Configuration Bus Display - Jelastic
# ===========================================
# Copiez ce fichier vers .env.local et modifiez les valeurs
# cp env.example.txt .env.local

# ===========================================
# Configuration Serveur (OBLIGATOIRE)
# ===========================================

# Environnement (development, production)
NODE_ENV=production

# Port de l'application
PORT=3000

# Adresse d'écoute (0.0.0.0 pour accepter toutes les connexions)
# IMPORTANT pour Jelastic: doit être 0.0.0.0
HOSTNAME=0.0.0.0

# ===========================================
# Configuration Redis (RECOMMANDÉ en production)
# ===========================================

# Activer Redis (true/false)
# Si false, utilise le cache Map local (perdu au redémarrage)
REDIS_ENABLED=true

# URL de connexion Redis avec authentification
# Format: redis://[user]:[password]@[host]:[port]
# Exemple Jelastic: redis://admin:VOTRE_MOT_DE_PASSE@10.100.XX.XX:6379
REDIS_URL=redis://admin:VOTRE_MOT_DE_PASSE@10.100.XX.XX:6379

# Préfixe des clés Redis (pour éviter les collisions)
REDIS_PREFIX=bus-display:

# ===========================================
# Configuration Cache
# ===========================================

# Durée du cache en millisecondes (défaut: 45000 = 45s)
# Les données de transport sont mises en cache pendant cette durée
CACHE_TTL=45000

# ===========================================
# Sécurité (OBLIGATOIRE en production)
# ===========================================

# Clé API pour accès admin et métriques
# IMPORTANT: Changez cette valeur en production!
# Générer avec: openssl rand -hex 32
ADMIN_API_KEY=CHANGEZ_CETTE_CLE_EN_PRODUCTION

# ===========================================
# Monitoring
# ===========================================

# Activer les métriques Prometheus (/api/metrics)
METRICS_ENABLED=true

# Désactiver la télémétrie Next.js
NEXT_TELEMETRY_DISABLED=1
